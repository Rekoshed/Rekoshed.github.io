Метод style

простой метод стилизации

Здесь в качестве примера мы берём строчку текста, и стилизуем её применяя java script. 

<div id="Setting">Настройка изменений...ждите</div>
<script>
// получаем html элемент по его идентефикатору
var info = &document.getElementById("Setting");
// вносим содержимое элемента
info.innerHTML = "Изменения HTML контента работают!";
// вносим цвет содержимого
info.style.color = "Red";
// устанавливаем шрифт внутри элемента
inf.style.fontSize = "large";
</script>
Настройка изменений...ждите

Применяем функции

// получаем элемент
var elem = document.getElementById("Settingll");
// определяем функцию для содержимого. в нашем случае - текст
function text( el, str ) {
if ( el.textContent ) {
el.textContent = str;
} else {
el.innerText = str;
}
}

function size ( el, str ) {
el.style.fontSize = str;
}

function color ( el, str ) {
el.style.color = str;
}

size( elem, '10px') 
color( elem, 'Darkblue' )
text(elem, 'Функции к стилю применены!!!')
